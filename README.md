# Supported OS/Distros
  - Debian GNU/Linux
  - Ubuntu
  - Fedora
  - CentOS
  - Dyson
  - Manjaro
  - ALT Linux
  - FreeBSD
  - openSUSE
  - Solaris

# Dependencies
  - make (during install, update, uninstall time only)
  - git (during update time only)

# Install lctl
  - `git clone https://github.com/larkvirtual/lctl.git`
  - `cd lctl`
  - `sudo make install`

# Uninstall lctl
  - `cd lctl`
  - `sudo make uninstall`

# Update lctl
  - `cd lctl`
  - `make update`

# Use lctl
  - `lctl .ssh`
  - `lctl .gitconfig`
  - `sudo lctl sources.list`
